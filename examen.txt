create table EMPLOYEES(
EMPLOYEES_PK int primary key not null,

DATE_OF_JOININD  Informes de Modelador de Datos

);


create table PROJECTS(
PROJECTS_PK int primary key not null,

BUDGET DECIMAL(10,2) not null,
DATE_OF_BIGINNING DATE,

);
